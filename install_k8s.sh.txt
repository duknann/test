#!/bin/bash
set -e

echo "[1/9] Cập nhật hệ thống và cài các gói cần thiết..."
sudo apt update && sudo apt install -y apt-transport-https curl

echo "[2/9] Thêm kho lưu trữ Kubernetes và cập nhật..."
curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo apt-key add -
echo "deb https://apt.kubernetes.io/ kubernetes-xenial main" | sudo tee /etc/apt/sources.list.d/kubernetes.list
sudo apt update

echo "[3/9] Cài đặt kubelet, kubeadm, kubectl và kubernetes-cni..."
sudo apt install -y kubelet kubeadm kubectl kubernetes-cni
sudo apt-mark hold kubelet kubeadm kubectl

echo "[4/9] Vô hiệu hóa swap..."
sudo swapoff -a
sudo sed -i '/ swap / s/^/#/' /etc/fstab

echo "[5/9] Cấu hình iptables để xem lưu lượng được bắc cầu..."
sudo modprobe br_netfilter
echo 'net.bridge.bridge-nf-call-iptables=1' | sudo tee /etc/sysctl.d/k8s.conf
sudo sysctl --system

echo "[6/9] Cài đặt Docker và cấu hình cgroup driver là systemd..."
sudo apt install -y docker.io
sudo mkdir -p /etc/docker
cat <<EOF | sudo tee /etc/docker/daemon.json
{
  "exec-opts": ["native.cgroupdriver=systemd"],
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "100m"
  },
  "storage-driver": "overlay2"
}
EOF
sudo systemctl enable docker
sudo systemctl daemon-reexec
sudo systemctl restart docker

echo "[7/9] Khởi tạo Kubernetes Master Node..."
sudo kubeadm init --pod-network-cidr=10.244.0.0/16

echo "[8/9] Cấu hình kubeconfig cho user hiện tại..."
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

echo "[9/9] Triển khai mạng Pod với Flannel..."
kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/kube-flannel.yml
kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/k8s-manifests/kube-flannel-rbac.yml

echo "✅ Hoàn tất cài đặt Kubernetes theo hướng dẫn của Bizfly Cloud!"
kubectl get nodes
